    function updateInventoryStats(data) {
      let aman = 0, kurang = 0, kritis = 0;
      for (let i = 0; i < data.length; i++) {
        const s = String(data[i][9] || "").trim().toUpperCase();
        if (s === "AMAN") aman++;
        else if (s === "KURANG") kurang++;
        else if (s === "KRITIS") kritis++;
      }
      const elAman = document.getElementById('countAman');
      const elKurang = document.getElementById('countKurang');
      const elKritis = document.getElementById('countKritis');
      if(elAman) elAman.innerText = aman;
      if(elKurang) elKurang.innerText = kurang;
      if(elKritis) elKritis.innerText = kritis;
    }

    function renderTable(bodyId, data) {
      const tbody = document.getElementById(bodyId);
      if (!tbody) return;

      if (bodyId === 'bodyInv') {
        document.getElementById('totalInv').innerText = data.length;
        
        // Logika Sinkronisasi: Inactive Item = Stok Kritis
        document.getElementById('inactiveInv').innerText = data.filter(r => String(r[9] || "").trim().toUpperCase() === "KRITIS").length;
        
        // MODIFIKASI: Ambil 20 data terakhir dari database dan urutkan yang terbaru di atas
        const limited = data.slice(-20).reverse(); 
        
        tbody.innerHTML = limited.map(r => {
          const statusDariSheet = (r[9] || "").toUpperCase();
          let kls = (statusDariSheet === "AMAN") ? "bg-aman" : (statusDariSheet === "KURANG") ? "bg-kurang" : (statusDariSheet === "KRITIS") ? "bg-kritis" : "";
          
// --- MODIFIKASI IKON DI SINI ---
      return `<tr>
        <td>${r[0] || ""}</td>
        <td>${r[1] || ""}</td>
        <td style="font-weight:bold;">${r[2] || 0}</td>
        <td>${r[3] || ""}</td>
        <td><span class="badge ${kls}">${statusDariSheet}</span></td>
        <td>
          <div style="display: flex; gap: 12px; align-items: center;">
            <!-- Ikon Edit Modern -->
            <span class="material-symbols-outlined icon-btn edit-color" 
                  onclick='openEditModal(${JSON.stringify(r)})' 
                  title="Edit">edit_square</span>
            
            <!-- Ikon Delete Modern -->
            <span class="material-symbols-outlined icon-btn delete-color" 
                  onclick="confirmDelete('${r[0]}')" 
                  title="Hapus">delete</span>
          </div>
        </td>
      </tr>`;
    }).join('');
        
        // Update keterangan di bawah tabel agar sesuai
        const note = tbody.closest('.card').querySelector('p.italic');
        if(note) note.innerText = "* Menampilkan 20 data terbaru dari baris terakhir";

      } else {
        // ... bagian Info PO tetap sama seperti sebelumnya ...
        const currentYear = "2026";
        const validPO = data.filter(r => r[0]);
        let totalVal = validPO.reduce((a, b) => a + (parseFloat(String(b[12] || "0").replace(/[^0-9.-]+/g, "")) || 0), 0);
        let val2026 = allPO.reduce((a, b) => a + (String(b[1]).includes(currentYear) ? (parseFloat(String(b[12] || "0").replace(/[^0-9.-]+/g, "")) || 0) : 0), 0);
        document.getElementById('totalPO').innerText = [...new Set(validPO.map(r => r[0]))].length;
        document.getElementById('netValueTotal').innerText = totalVal.toLocaleString('id-ID');
        if(document.getElementById('netValueYear')) document.getElementById('netValueYear').innerText = val2026.toLocaleString('id-ID');
        if(allPO.length > 0) {
          const allDates = allPO.map(r => r[1]).filter(d => d);
          document.getElementById('lastUpdateText').innerText = "Update Terakhir Data: " + allDates[allDates.length-1];
        }
        const limited = validPO.slice(-20).reverse();
        tbody.innerHTML = limited.map(r => `<tr>${r.map(c => `<td>${c || ""}</td>`).join('')}</tr>`).join('');
      }
    }

    function filterTable(inputId, bodyId, sourceData) {
      clearTimeout(filterTimeout);
      filterTimeout = setTimeout(() => {
        const q = document.getElementById(inputId).value.toLowerCase();
        const yf = document.getElementById('filterTahun')?.value || "All";
        const res = sourceData.filter(r => {
          const mSearch = r.some(c => String(c).toLowerCase().includes(q));
          const mYear = (bodyId === 'bodyPO' && yf !== "All") ? String(r[1]).includes(yf) : true;
          return mSearch && mYear;
        });
        renderTable(bodyId, res);
      }, 300);
    }

    async function startScanner() {
  const reader = document.getElementById('reader');
  const btnScan = document.querySelector('.btn-scan-modern');
  
  // Tampilkan Notifikasi Proses
  launchToast("Membuka kamera...");
  
  // Efek Loading pada tombol
  if(btnScan) btnScan.innerHTML = `<span class="material-symbols-outlined spin">sync</span> <span>Wait..</span>`;

  try {
    reader.style.display = 'block';
    
    if (html5QrCode) { 
      await html5QrCode.stop().catch(()=>{}); 
      html5QrCode = null; 
    }
    
    html5QrCode = new Html5Qrcode("reader");
    
    await html5QrCode.start(
      { facingMode: "environment" }, 
      { fps: 10, qrbox: 250 }, 
      (text) => {
        document.getElementById('searchBar').value = text;
        filterTable('searchBar', 'bodyInv', allInv);
        launchToast("Material Ditemukan!"); // Notif Sukses Scan
        stopScanner();
        // Kembalikan tombol scan
        if(btnScan) btnScan.innerHTML = `<span class="material-symbols-outlined">qr_code_scanner</span> <span>Scan</span>`;
      }
    ).catch(err => {
       throw new Error("Kamera diblokir atau tidak ditemukan.");
    });

  } catch (err) {
    // Notifikasi Gagal yang Kekinian
    launchToast(err.message, "error");
    reader.style.display = 'none';
    if(btnScan) btnScan.innerHTML = `<span class="material-symbols-outlined">qr_code_scanner</span> <span>Scan</span>`;
  }
}

    function launchToast(msg, type = 'success') {
  const container = document.getElementById('toast-container');
  const toast = document.createElement('div');
  toast.className = `toast ${type === 'error' ? 'error' : ''}`;
  
  const icon = type === 'success' ? '✅' : '❌';
  toast.innerHTML = `<span>${icon}</span><span class="toast-msg">${msg}</span>`;
  
  container.appendChild(toast);
  
  // Animasi masuk
  setTimeout(() => toast.classList.add('show'), 100);
  
  // Hapus otomatis setelah 3 detik
  setTimeout(() => {
    toast.classList.remove('show');
    setTimeout(() => toast.remove(), 500);
  }, 3000);
}
